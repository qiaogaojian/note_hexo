# Unity中自带数据固化的属性

```C#
    private bool guideFinish = false;

    public bool GuideFinish
    {
        get
        {
            if (PlayerPrefs.HasKey("guideFinish"))
            {
                guideFinish = PlayerPrefs.GetInt("guideFinish") == 1 ? true : false;
            }
            else
            {
                PlayerPrefs.SetInt("guideFinish", guideFinish ? 1 : 0);
            }
            return guideFinish;
        }

        set
        {
            guideFinish = value;
            if (guideFinish)
            {
                PlayerPrefs.SetInt("guideFinish",1);
            }
            else
            {
                PlayerPrefs.SetInt("guideFinish",0);
            }
        }
    }
```
